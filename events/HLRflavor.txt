country_event = {
	id = hlrmod_flavor_event.6
	title = hlrmod_flavor_event.6.title
	desc = hlrmod_flavor_event.6.desc
	picture = HLR_DUTCHINDEP_eventPicture
	is_triggered_only = yes

	#release
	option = {
		name = hlrmod_flavor_event.6.a
		if { limit = { exists = HOL }
			ROOT = {
				96 = { 
					cede_province = HOL
					remove_core = HLR
				}
				97 = { 
					cede_province = HOL
					remove_core = HLR
				}
				4383 = { 
					cede_province = HOL
					remove_core = HLR
				}
			}
		}
		else {
			custom_tooltip = releasing_holland.tooltip
			hidden_effect = {
				every_province = {
					remove_core = HOL
				}
				96 = { 
					add_core = HOL 
					remove_core = HLR
				}
				97 = { 
					add_core = HOL 
					remove_core = HLR
				}
				4383 = { 
					add_core = HOL 
					remove_core = HLR
				}
				ROOT = { release = HOL }
			}
		}
		create_guarantee = HOL
		add_historical_friend = HOL
		custom_tooltip = dutch_wealth.tootip
		hidden_effect = {
			every_owned_province = {
				limit = {
					region = low_countries_region
				}
				remove_province_modifier = hlr_dutch_riots
				add_province_modifier = {
					name = hlr_dutch_wealth
					duration = 7300
				}
			}
		}
		hidden_effect = {
			clr_country_flag = lowland_focus
			set_country_flag = released_the_dutch
		}
	}

	#Reichsherzogtum
	option = {
		name = hlrmod_flavor_event.6.b
		custom_tooltip = hlr_reichsherzogtum_holland.tooltip
		hidden_effect = {
			every_province = {
				remove_core = HOL
			}
			96 = { 
				remove_core = HLR
				add_core = HOL 
			}
			97 = { 
				remove_core = HLR
				add_core = HOL 
			}
			4383 = { 
				remove_core = HLR
				add_core = HOL 
			}
			ROOT = { release = HOL }
			HOL = { set_government_rank = 1 }
			create_subject = {
				subject = HOL
				subject_type = imperialduchy
			}
			clr_country_flag = lowland_focus
			set_country_flag = released_the_dutch
			
		}	
	}

	#keep them
	option = {
		name = hlrmod_flavor_event.6.c
		clr_country_flag = lowland_focus
		custom_tooltip = dutch_anger.tooltip
		hidden_effect = {
			every_owned_province = {
				limit = {
					region = low_countries_region
				}
				add_province_modifier = {
					name = hlr_dutch_anger

				}
			}
		}

	}
