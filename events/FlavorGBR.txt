country_event = {
	id = flavor_gbr.7
	title = flavor_gbr.7.t
	desc = flavor_gbr.7.desc
	picture = FUR_TRADE_eventPicture
	is_triggered_only = yes
	major = yes
	#Nice
	option = {
		name = flavor_gbr.7.a
		if = {
			limit = {
				NOT = { has_global_flag = had_fur_trade_event }
			}
			add_treasury = 500
			change_price = {
				trade_goods = fur
				key = FUR_TRADE
				value = 0.25
				duration = 3650
			}
		}
		else = {
			add_treasury = 2000
		}
		ai_chance = {
			factor = 1
		}
	}
}
