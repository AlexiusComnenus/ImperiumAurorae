			limit = {
				capital_scope = { region = iberia_region }
				culture_group = iberian
			}

				inherit = PREV
			}
		}
		
		country_event = { id = ia_flavor_val.3 days = 5}
		
		TEM = { country_event = { id = ia_flavor_tem.7 days = 5} }
	}
}
