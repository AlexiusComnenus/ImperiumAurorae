		add_stability = -3

			country_event = { id = saxon_crisis.1  days = 500 }

				limit = { NOT = { is_in_war = { casus_belli = cb_german_civil_war } } }

			}
			every_subject_country = {
				limit = { is_subject_of_type = german_march }
				ROOT = {
					create_subject = {
						subject_type = saxon_daimyo
						subject = PREV
					}
				}
			}
			every_subject_country = {
				limit = {
					is_subject_of_type = saxon_daimyo
				}
				every_war_enemy_country = {
					limit = { NOT = { is_in_war = { casus_belli = cb_german_civil_war } } }
					white_peace = PREV
				}
			}
			if = {
				limit = { ai = yes }
				every_subject_country = {
					limit = {
						is_subject_of_type = tributary_state
						culture_group = germanic
					}
					ROOT = {
						create_subject = {
							subject_type = saxon_daimyo

					}
